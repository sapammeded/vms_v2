function doPost(e) {

function handleRequest(e, method) {
  // Inisialisasi output JSON di awal
  var output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);
  
  try {
    var params = e.parameter || {};
    var path = e.pathInfo || "";
    
    // Penanganan khusus untuk POST JSON body
    if (method === "POST" && e.postData && e.postData.contents) {
      try {
        var jsonBody = JSON.parse(e.postData.contents);
        for (var key in jsonBody) {
          params[key] = jsonBody[key];
        }
      } catch(parseErr) {
        // Abaikan jika bukan JSON
      }
    }
    
    // --- FIX TOKEN: Pastikan token dari URL parameter masuk ke params ---
    if (e.parameter.token) {
      params.token = e.parameter.token;
    }
    
    // Panggil routeRequest dengan parameter yang sudah dibersihkan
    var result = routeRequest(path, params, method);
    
    output.setContent(JSON.stringify(result));
    return output;
    
  } catch (err) {
    // Tangkap semua error sistem agar dashboard tidak hang
    output.setContent(JSON.stringify({ 
      ok: false, 
      error: "System Error: " + err.toString() 
    }));
    return output;
  }
}

function routeRequest(path, params, method) {
  // PENTING: Gunakan actualPath agar bisa membaca parameter dari URL (?path=...)
  var actualPath = path || params.path || ""; 
  
  // 1. LOGIN
  if (actualPath === "/login") {
    return doLogin(params);
  }
  
  if (!params.token || !validateToken(params.token)) {
    return { ok: false, error: "Unauthorized: Token tidak valid atau kadaluarsa" };
  }

  // 2. DASHBOARD STATS
  if (actualPath === "/admin/stats") {
    return getStats();
  }
  
  // 3. COMPANIES (Daftar Perusahaan)
  if (actualPath === "/admin/companies") {
    return getCompanies();
  }
  
  // 4. COMPANY DETAIL
  if (actualPath && actualPath.indexOf("/admin/company/") === 0) {
    var companyId = actualPath.split("/").pop();
    return getCompanyDetail(companyId);
  }
  
  // 5. DEVICES (Daftar Perangkat)
  if (actualPath === "/admin/devices") {
    return getDevices();
  }
  
  // 6. ACTIVITY / VIOLATIONS (Log Aktivitas)
  if (actualPath === "/admin/activity") {
    var limit = params.limit ? parseInt(params.limit) : 500;
    return getActivity(limit);
  }
  
  // 7. INVOICES
  if (actualPath === "/admin/invoices") {
    return getInvoices();
  }
  
  // 8. DEVICE REQUESTS
  if (actualPath === "/admin/device-requests") {
    var status = params.status || "";
    return getDeviceRequests(status);
  }
  
  // 9. LICENSE ACTIONS (POST/GET dialihkan ke actualPath)
  if (actualPath === "/generate-license") return generateLicense(params);
  if (actualPath === "/renew-license") return renewLicense(params);
  if (actualPath === "/update-package") return updatePackage(params);
  if (actualPath === "/approve-device") return approveDevice(params);
  if (actualPath === "/approve-device-request") return approveDeviceRequest(params);
  if (actualPath === "/delete-device") return deleteDevice(params);
  if (actualPath === "/delete-company") return deleteCompany(params);
  if (actualPath === "/mark-invoice-paid") return markInvoicePaid(params);
  
  // 10. SETTINGS & SYSTEM
  if (actualPath === "/admin/settings") return saveSettings(params);
  if (actualPath === "/admin/reset-system") return resetSystem();
  if (actualPath === "/admin/users") return getAdminUsers();
  if (actualPath === "/admin/add-user") return addAdminUser(params);
  if (actualPath === "/admin/delete-user") return deleteAdminUser(params);
  if (actualPath === "/admin/change-password") return changePassword(params);
  
  // 11. CRON JOB
  if (actualPath === "/cron/check-expired") return checkExpiredLicenses();
  
  // Jika tidak ada yang cocok
  return { ok: false, error: "Endpoint tidak ditemukan: " + actualPath };
}


function validateToken(token) {
  try {
    var raw = Utilities.newBlob(Utilities.base64Decode(token)).getDataAsString();
    var parts = raw.split(":");
    if (parts.length < 2) return false;
    var issuedAt = parseInt(parts[1], 10);
    if (!issuedAt || isNaN(issuedAt)) return false;
    var maxAgeMs = 24 * 60 * 60 * 1000;
    return (Date.now() - issuedAt) <= maxAgeMs;
  } catch (e) {
    return false;
  }
}

// ================= AUTHENTICATION =================
function doLogin(params) {
  var user = params.username;
  var pass = params.password;
  
  if (!user || !pass) {
    return { ok: false, error: "Username dan Password tidak boleh kosong" };
  }
  
  var sheet = getUserSheet(); 
  var data = sheet.getDataRange().getValues();
  
  for (var i = 1; i < data.length; i++) {
    var dbUser = data[i][0];
    var dbPassEncrypted = data[i][1];
    var dbRole = data[i][2];
    
    // Decode password dari Base64 di sheet agar bisa dibandingkan dengan input
    var decodedDbPass = "";
    try {
      decodedDbPass = Utilities.newBlob(Utilities.base64Decode(dbPassEncrypted)).getDataAsString();
    } catch(e) {
      decodedDbPass = dbPassEncrypted; // Jika format di sheet bukan Base64, anggap teks biasa
    }
    
    if (dbUser == user && decodedDbPass == pass) {
      var token = Utilities.base64Encode(user + ":" + Date.now());
      return { 
        ok: true, 
        token: token, 
        username: dbUser, 
        role: dbRole
      };
    }
  }
  
  return { ok: false, error: "Username atau Password salah!" };
}


function changePassword(params) {
  var username = params.username;
  var oldPassword = params.oldPassword;
  var newPassword = params.newPassword;
  if (!username || !oldPassword || !newPassword) return { ok: false, error: "Parameter tidak lengkap" };
  var sheet = getUserSheet();
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] != username) continue;
    var current = "";
    try { current = Utilities.newBlob(Utilities.base64Decode(data[i][1])).getDataAsString(); } catch(e) { current = data[i][1]; }
    if (current != oldPassword) return { ok: false, error: "Password lama salah" };
    sheet.getRange(i + 1, 2).setValue(Utilities.base64Encode(newPassword));
    sheet.getRange(i + 1, 4).setValue(new Date().toISOString());
    return { ok: true, message: "Password berhasil diubah" };
  }
  return { ok: false, error: "User tidak ditemukan" };
}

// ================= STATS =================
function getStats() {
  var companies = getCompaniesData();
  var devices = getDevicesData();
  var activity = getActivityData();
  var invoices = getInvoicesData();
  
  var now = Date.now();
  var activeCompanies = 0;
  for (var i = 0; i < companies.length; i++) {
    if (companies[i].expiredAt > now && companies[i].status === "ACTIVE") activeCompanies++;
  }
  
  var activeDevices = 0, pendingDevices = 0, suspendedDevices = 0;
  for (var j = 0; j < devices.length; j++) {
    if (devices[j].status === "ACTIVE") activeDevices++;
    else if (devices[j].status === "PENDING_APPROVAL") pendingDevices++;
    else if (devices[j].status === "SUSPENDED" || devices[j].status === "BANNED") suspendedDevices++;
  }
  
  var packageCount = { DEMO: 0, BASIC: 0, PRO: 0 };
  for (var k = 0; k < companies.length; k++) {
    var pkg = companies[k].package;
    if (packageCount[pkg] !== undefined) packageCount[pkg]++;
  }
  
  var violations = [];
  for (var l = 0; l < activity.length; l++) {
    if (activity[l].type === "VIOLATION_REPORTED") violations.push(activity[l]);
  }
  
  var nowTime = Date.now();
  var last30Days = nowTime - 30 * 24 * 60 * 60 * 1000;
  var violations30Days = 0, violations7Days = 0;
  for (var m = 0; m < violations.length; m++) {
    if (violations[m].timestamp > last30Days) violations30Days++;
    if (violations[m].timestamp > nowTime - 7 * 24 * 60 * 60 * 1000) violations7Days++;
  }
  
  var last30DaysRevenue = 0;
  for (var n = 0; n < invoices.length; n++) {
    if (invoices[n].status === "PAID" && invoices[n].paidAt > last30Days) {
      last30DaysRevenue += invoices[n].amount;
    }
  }
  
  return {
    ok: true,
    companies: {
      total: companies.length,
      active: activeCompanies,
      byPackage: packageCount
    },
    devices: {
      total: devices.length,
      active: activeDevices,
      pending: pendingDevices,
      suspended: suspendedDevices,
      banned: 0
    },
    violations: {
      total: violations.length,
      last30Days: violations30Days,
      last7Days: violations7Days
    },
    revenue: {
      last30Days: last30DaysRevenue
    }
  };
}

function getCompanies() {
  var companies = getCompaniesData();
  var now = Date.now();
  for (var i = 0; i < companies.length; i++) {
    if (companies[i].expiredAt < now && companies[i].status === "ACTIVE") {
      companies[i].status = "EXPIRED";
    }
  }
  return companies;
}

function getCompanyDetail(companyId) {
  var companies = getCompaniesData();
  var company = null;
  for (var i = 0; i < companies.length; i++) {
    if (companies[i].id === companyId) {
      company = companies[i];
      break;
    }
  }
  
  if (!company) {
    return { ok: false, error: "Company not found" };
  }
  
  var devices = getDevicesData();
  var companyDevices = [];
  for (var j = 0; j < devices.length; j++) {
    if (devices[j].companyId === companyId) companyDevices.push(devices[j]);
  }
  
  return {
    id: company.id,
    companyName: company.companyName,
    licenseKey: company.licenseKey,
    package: company.package,
    pic: company.pic,
    phone: company.phone,
    email: company.email,
    address: company.address,
    maxDevices: company.maxDevices,
    expiredAt: company.expiredAt,
    status: company.status,
    createdAt: company.createdAt,
    notes: company.notes,
    stats: { totalDevices: companyDevices.length },
    devices: companyDevices
  };
}

function generateLicense(params) {
  var sheet = getCompanySheet();
  var now = new Date();
  var expiredAt = new Date();
  
  var pkg = params.package || "DEMO";
  var maxDevices = params.customMaxDevices ? parseInt(params.customMaxDevices) : (pkg === "PRO" ? 999 : (pkg === "BASIC" ? 10 : 2));
  
  switch (pkg) {
    case "DEMO": expiredAt.setDate(expiredAt.getDate() + 7); break;
    case "BASIC": expiredAt.setMonth(expiredAt.getMonth() + 1); break;
    case "PRO": expiredAt.setMonth(expiredAt.getMonth() + 1); break;
  }
  
  var licenseKey = "VMS-" + Utilities.getUuid().substring(0, 8).toUpperCase();
  var companyId = "cmp_" + Date.now() + "_" + Math.floor(Math.random() * 10000);
  
  sheet.appendRow([
    companyId,
    params.companyName || "-",
    licenseKey,
    pkg,
    params.pic || "-",
    params.phone || "-",
    params.email || "-",
    params.address || "",
    maxDevices,
    0,
    expiredAt.toISOString(),
    "ACTIVE",
    now.toISOString(),
    params.notes || ""
  ]);
  
  SpreadsheetApp.flush();
  
  return {
    ok: true,
    licenseKey: licenseKey,
    companyId: companyId
  };
}

function renewLicense(params) {
  var sheet = getCompanySheet();
  var companies = getCompaniesData();
  var companyId = params.companyId;
  var months = parseInt(params.months) || 1;
  var amount = parseInt(params.amount) || 0;
  
  var companyIndex = -1;
  var companyData = null;
  for (var i = 0; i < companies.length; i++) {
    if (companies[i].id === companyId) {
      companyIndex = i;
      companyData = companies[i];
      break;
    }
  }
  
  if (companyIndex === -1) {
    return { ok: false, error: "Company not found" };
  }
  
  var oldExpiry = new Date(companyData.expiredAt);
  var newExpiry = new Date(oldExpiry);
  newExpiry.setMonth(newExpiry.getMonth() + months);
  
  var row = companyIndex + 2;
  sheet.getRange(row, 11).setValue(newExpiry.toISOString());
  sheet.getRange(row, 12).setValue("ACTIVE");
  SpreadsheetApp.flush();
  
  var invoiceSheet = getInvoiceSheet();
  var invoiceId = "INV_" + Date.now() + "_" + Math.floor(Math.random() * 10000);
  invoiceSheet.appendRow([
    invoiceId,
    companyId,
    companyData.companyName,
    amount,
    months,
    companyData.package,
    amount > 0 ? "UNPAID" : "PAID",
    new Date().toISOString(),
    amount > 0 ? "" : new Date().toISOString()
  ]);
  
  return { ok: true };
}

  var approve = params.approve === "true" || params.approve === true;
  var notes = params.notes || "";
  
  var requests = getDeviceRequests("");
  var requestIndex = -1;
  for (var i = 0; i < requests.length; i++) {
    if (requests[i].id === requestId) {
      requestIndex = i;
      break;
    }
  }
  
  if (requestIndex === -1) {
    return { ok: false, error: "Request not found" };
  }
  
  var row = requestIndex + 2;
  
  if (approve) {
    var fee = requests[requestIndex].fee;
    sheet.getRange(row, 7).setValue("WAITING_PAYMENT");
    
    var invoiceSheet = getInvoiceSheet();
    var invoiceId = "INV_" + Date.now() + "_" + Math.floor(Math.random() * 10000);
    invoiceSheet.appendRow([
      invoiceId,
      requests[requestIndex].companyId,
      requests[requestIndex].companyName,
      fee,
      1,
      "EXTRA_DEVICE",
      "UNPAID",
      new Date().toISOString(),
      ""
    ]);
    
    SpreadsheetApp.flush();
    
    return {
      ok: true,
      invoiceId: invoiceId,
      amount: fee
    };
  } else {
    sheet.getRange(row, 7).setValue("REJECTED");
    sheet.getRange(row, 9).setValue(notes);
    SpreadsheetApp.flush();
    return { ok: true };
  }
}

// ================= ADMIN USERS =================
function getAdminUsers() {
  var sheet = getUserSheet();
  var lastRow = sheet.getLastRow();
  if (lastRow <= 1) return [];
  
  var data = sheet.getRange(2, 1, lastRow - 1, 3).getValues();
  var users = [];
  for (var i = 0; i < data.length; i++) {
    users.push({
      username: data[i][0],
      role: data[i][2]
    });
  }
  return users;
}

function addAdminUser(params) {
  var sheet = getUserSheet();
  var username = params.username;
  var password = params.password;
  var role = params.role || "ADMIN";
  
  var existing = getAdminUsers().filter(function(u) { return u.username === username; });
  if (existing.length > 0) {
    return { ok: false, error: "Username already exists" };
  }
  
  sheet.appendRow([
    username,
    btoa(password),
    role,
    new Date().toISOString()
  ]);
  
  SpreadsheetApp.flush();
  return { ok: true };
}

function deleteAdminUser(params) {
  var sheet = getUserSheet();
  var username = params.username;
  
  if (username === "admin") {
    return { ok: false, error: "Cannot delete default admin" };
  }
  
  var users = getAdminUsers();
  var userIndex = -1;
  for (var i = 0; i < users.length; i++) {
    if (users[i].username === username) {
      userIndex = i;
      break;
    }
  }
  
  if (userIndex === -1) {
    return { ok: false, error: "User not found" };
  }
  
  var row = userIndex + 2;
  sheet.deleteRow(row);
  SpreadsheetApp.flush();
  
  return { ok: true };
}

// ================= SETTINGS =================
function saveSettings(params) {
  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  var sheet = ss.getSheetByName("vms_settings");
  if (!sheet) {
    sheet = ss.insertSheet("vms_settings");
    sheet.appendRow(["Key", "Value"]);
  }
  
  var pricing = params.pricing;
  var general = params.general;
  
  if (pricing) {
    setSetting(sheet, "price_basic", pricing.BASIC ? pricing.BASIC.price : 500000);
    setSetting(sheet, "price_pro", pricing.PRO ? pricing.PRO.price : 2000000);
    setSetting(sheet, "extra_device_fee", pricing.BASIC ? pricing.BASIC.extraDeviceFee : 50000);
  }
  
  if (general) {
    setSetting(sheet, "tax_rate", general.tax || 11);
  }
  
  SpreadsheetApp.flush();
  return { ok: true };
}

function setSetting(sheet, key, value) {
  var lastRow = sheet.getLastRow();
  var data = sheet.getRange(1, 1, lastRow, 2).getValues();
  for (var i = 0; i < data.length; i++) {
    if (data[i][0] === key) {
      sheet.getRange(i + 1, 2).setValue(value);
      return;
    }
  }
  sheet.appendRow([key, value]);
}

// ================= SYSTEM =================
function checkExpiredLicenses() {
  var sheet = getCompanySheet();
  var companies = getCompaniesData();
  var now = Date.now();
  var updated = 0;
  
  for (var i = 0; i < companies.length; i++) {
    if (companies[i].expiredAt < now && companies[i].status === "ACTIVE") {
      var row = i + 2;
      sheet.getRange(row, 12).setValue("EXPIRED");
      updated++;
    }
  }
  
  SpreadsheetApp.flush();
  return { ok: true, updated: updated > 0 };
}

function resetSystem() {
  var sheets = ["vms_license", "vms_devices", "vms_activity", "vms_invoices", "vms_device_requests"];
  for (var s = 0; s < sheets.length; s++) {
    var sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(sheets[s]);
    if (sheet) {
      var lastRow = sheet.getLastRow();
      if (lastRow > 1) {
        sheet.deleteRows(2, lastRow - 1);
      }
    }
  }
  
  return { ok: true };
}
